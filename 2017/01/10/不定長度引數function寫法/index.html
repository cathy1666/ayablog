不定長度引數函數的寫法

Posted on 2017-01-10

需求

把開 特定檔案、寫入帶有當下時間的log、關檔案 整個流程寫成一個function，且此function要有像printf的介面，可以自動偵測印出含有任意數量的變數的string。

Sample Code

int printLog(char* hex, int len, char* char_p, ...)
{
    int i;
    FILE *fp;
    unsigned char* hex_p = (unsigned char*)hex;

    fp = fopen(LOGFILE, "a");
    if (NULL == fp)
    {
        printf("open %s fail\n", LOGFILE);
        return -1;
    }    

    char buffer[256];
    va_list list_p;
    va_start(list_p, char_p);
    vsprintf(buffer, char_p, list_p);
    va_end(list_p);

    //print time stamp
    fprintf(fp, "%.24s   ", getTime()); 

    //print comment with variable-length argument 
    fprintf(fp, "%s", buffer);

    //print hex string    
    fprintf(fp, "[ ");
    for (i = 0; i < len; i++)
    {
        fprintf(fp, "%02x ", *hex_p++);
    }
    fprintf(fp, "]\n");

    fclose(fp);
    return 0;    
}
